# 플레이스 분석 — 수집 서버 워커 (참고 코드)

> **모델: 운영자(나) 수집 서버 1대**가 모든 사장님의 분석요청을 처리한다. (사장님은 버튼만 누름 — 각자 PC 안 켬)
> 지금은 그 1대가 내 집 PC/미니PC일 뿐. 사장님 늘면 미니PC 추가 or 클라우드+레지덴셜 프록시로 이전.
>
> 와벨리 [🔍 분석 시작] 버튼이 실제로 동작하려면 이 수집 서버에 아래 워커가 돌아야 한다.
> 와벨리는 요청만 큐에 넣고, 수집(스크래핑·N지수 계산)은 **가정용 IP 머신에서만** 가능하기 때문(네이버가 클라우드/데이터센터 IP 차단).
>
> ⚠️ 이건 **참고 글루 코드**다. 실제 스크래핑/N지수 계산은 너의 기존 수집기 함수를 호출하면 된다
> (`analyze_place(place_id)` 부분). 이 파일은 와벨리 repo에 두는 명세이고, 코드는 260616 수집기 폴더에 둘 것.

## 2) 워커 루프 (stdlib urllib 기반 — 새 의존성 없음)

```python
# place_collector/scheduler/wavely_worker.py
import os, json, time, urllib.request, urllib.parse

SB_URL = os.environ["SUPABASE_URL"]
SB_KEY = os.environ["SUPABASE_SERVICE_ROLE_KEY"]
HEADERS = {
    "apikey": SB_KEY,
    "Authorization": f"Bearer {SB_KEY}",
    "Content-Type": "application/json",
}

def _req(method, path, params=None, body=None):
    url = f"{SB_URL}/rest/v1/{path}"
    if params:
        url += "?" + urllib.parse.urlencode(params)
    data = json.dumps(body).encode() if body is not None else None
    r = urllib.request.Request(url, data=data, method=method, headers=HEADERS)
    with urllib.request.urlopen(r, timeout=30) as resp:
        txt = resp.read().decode()
        return json.loads(txt) if txt else None

def claim_pending():
    rows = _req("GET", "place_analysis_requests", {
        "status": "eq.pending", "order": "requested_at.asc", "limit": "1",
        "select": "id,store_id",
    })
    if not rows:
        return None
    req = rows[0]
    # running 으로 선점
    _req("PATCH", "place_analysis_requests", {"id": f"eq.{req['id']}"},
         {"status": "running"})
    return req

def store_place_id(store_id):
    rows = _req("GET", "stores", {"id": f"eq.{store_id}", "select": "naver_place_id"})
    return rows[0]["naver_place_id"] if rows else None

def finish(req_id, status, error=None):
    body = {"status": status, "finished_at": "now()"}
    if error:
        body["error"] = str(error)[:500]
    _req("PATCH", "place_analysis_requests", {"id": f"eq.{req_id}"}, body)

def push_results(store_id, place_id, result):
    """result = 너의 기존 수집기 분석 결과(아래 형태로 맞춰서 넘겨줄 것)"""
    today = time.strftime("%Y-%m-%d")
    # 2-1) 키워드별 순위 → place_rankings upsert (store_id,keyword,snap_date 유니크)
    rows = [{
        "store_id": store_id, "keyword": k["keyword"], "rank": k.get("rank"),
        "save_count": result.get("save_count"),
        "visitor_review": result.get("visitor_review"),
        "blog_review": result.get("blog_review"),
        "snap_date": today,
    } for k in result["keywords"]]
    if rows:
        # 충돌 시 갱신: Prefer resolution=merge-duplicates
        h = dict(HEADERS); h["Prefer"] = "resolution=merge-duplicates"
        req = urllib.request.Request(
            f"{SB_URL}/rest/v1/place_rankings?on_conflict=store_id,keyword,snap_date",
            data=json.dumps(rows).encode(), method="POST", headers=h)
        urllib.request.urlopen(req, timeout=30).read()
    # 2-2) 종합 스냅샷 → place_analysis insert
    _req("POST", "place_analysis", None, {
        "store_id": store_id, "place_id": place_id,
        "n1": result.get("n1"), "n2": result.get("n2"), "n3": result.get("n3"),
        "save_count": result.get("save_count"),
        "visitor_review": result.get("visitor_review"),
        "blog_review": result.get("blog_review"),
        "keyword_count": len(result["keywords"]),
        "exposed_count": sum(1 for k in result["keywords"] if k.get("rank")),
        "best_rank": min([k["rank"] for k in result["keywords"] if k.get("rank")], default=None),
        "avg_rank": (lambda rs: round(sum(rs)/len(rs), 2) if rs else None)(
                     [k["rank"] for k in result["keywords"] if k.get("rank")]),
    })

def loop():
    while True:
        try:
            req = claim_pending()
            if not req:
                time.sleep(15); continue
            place_id = store_place_id(req["store_id"])
            if not place_id:
                finish(req["id"], "failed", "연결된 네이버 플레이스 ID가 없어요"); continue
            # ↓↓↓ 여기서 너의 기존 수집기 종합분석 호출 (키워드 자동 발견 포함) ↓↓↓
            result = analyze_place(place_id)   # TODO: 기존 함수로 교체
            #   기대 형태:
            #   { "n1":..., "n2":..., "n3":...,
            #     "save_count":..., "visitor_review":..., "blog_review":...,
            #     "keywords": [ {"keyword":"춘천 닭갈비", "rank": 3}, ... ] }
            push_results(req["store_id"], place_id, result)
            finish(req["id"], "done")
            print(f"[wavely] done store={req['store_id']} kw={len(result['keywords'])}")
        except Exception as e:
            try:
                finish(req["id"], "failed", e)
            except Exception:
                pass
            print("[wavely] error:", e)
        time.sleep(3)  # RATE 보호

if __name__ == "__main__":
    loop()
```

## 3) ★ 스케줄 백그라운드 수집 — "바로 뜨는" 경험의 핵심 (필수)
adlog가 빠른 건 라이브 스크래핑이 아니라 **미리 모아둔 데이터를 보여주기 때문**이다. 그래서 이게 메인:
- **스케줄러로 N시간마다(예: 6시간), naver_place_id 가 있는 매장 전체를 백그라운드 수집**해서 `push_results()` 로 적재.
- 그러면 사장님이 화면 열면 **버튼 안 눌러도 최신 데이터가 1초 만에 뜬다**(와벨리 화면은 열 때 최신을 즉시 읽음).
- 온디맨드 [🔄 최신순위로 갱신] 버튼은 "지금 강제 새로고침"용 보조(1~2분). 일상은 스케줄 수집이 다 해줌.

```python
# 예: scheduler 에서 6시간마다 1회
def collect_all_registered():
    rows = _req("GET", "stores", {"naver_place_id": "not.is.null", "select": "id,naver_place_id"})
    for s in rows:
        try:
            result = analyze_place(s["naver_place_id"])   # 기존 종합분석 함수
            push_results(s["id"], s["naver_place_id"], result)
        except Exception as e:
            print("[wavely] schedule error", s["id"], e)
        time.sleep(5)  # 매장 간 간격 — RATE 보호
```
- 느린 건 "한 번도 안 모은 새 매장의 최초 1회"뿐. 그건 사장님이 [첫 분석 시작]을 누르면 온디맨드로 처리.
- RATE 한계: (매장수 × 키워드수)가 일일 3000 안에 들어와야 함. 넘으면 수집 주기를 늘리거나 미니PC 추가.

## 3.5) ★ 키워드 자동탐색 (D5 — 형 승인, 수집기 다음 우선작업)
현재 `wavely_sync.py` 의 `_analyze_store` 는 **등록 키워드(place_rank_keywords)만** 본다.
adlog처럼 **place_id만으로 노출 키워드 전체를 자동 발견**하게 만드는 게 D5의 목표.

구현 방향(수집기에서 택1 또는 혼합):
1. **place 상세에서 직접 추출** — 네이버 플레이스 상세/검색결과에 그 업체의 대표키워드·연관키워드가 노출됨. 그걸 긁어 후보로.
2. **카테고리+지역 조합 생성** — 업종(예: '오리고기')×지역(동/시) 조합으로 후보 키워드 생성 → 각 후보로 검색해 그 place_id 가 N위 안에 들면 채택. (adlog식: 노출되는 것만 남김)
3. **네이버 자동완성/연관검색 API** — 대표키워드 시드로 확장.

데이터 흐름(스키마 변경 없음):
- 자동발견 키워드 각각의 순위를 `place_rankings` 에 upsert (store_id, keyword, snap_date). → **와벨리 화면은 그대로 자동 표시됨**(목록·요약·급등급락·순위분포가 전부 place_rankings 기준이라 코드수정 불필요).
- `place_rank_keywords`(사용자 등록)는 그대로 두되 의미를 "**항상 고정으로 볼 키워드(선택)**" 로. 자동발견과 합집합(merge)으로 처리.
- 와벨리 UI 변경 필요: (선택) 등록 키워드 카드 문구를 "고정 키워드(선택) — 비워둬도 자동분석" 으로. (수집기 자동탐색 완성되면 알려주면 내가 2줄 바꿈)

RATE 주의: 자동탐색은 후보 검색이 많아 요청수가 늘어남. (매장수 × 후보키워드수)가 일일 3000 안에 들도록 후보 상한(예: 매장당 20~30)·수집주기 조절.

### ★ 구체 알고리즘 — "노출 키워드 역추적" (generate → verify)
```python
def discover_keywords(client, place_id, region, category, menu_terms=None, max_rank=30, cap=150):
    """place_id 가 노출되는 키워드를 후보생성→검증으로 역추적. 반환: [(keyword, rank), ...]"""
    # 1) 시드: 지역(시/동) × 업종/메뉴 × 공통수식어 조합
    regions = [region] + nearby_dongs(region)            # 춘천, 춘천 동면, 후평동 ...
    cats    = split_terms(category) + (menu_terms or [])  # 오리고기, 고기요리, 오리, 삼겹살 ...
    mods    = ['', '맛집', '추천', '회식', '단체', '모임', '야장', '점심', '저녁',
               '가족', '데이트', '룸', '주차', '1인', '예약', '근처']
    cand = set()
    for r in regions:
        for cz in cats:
            cand.add(f'{r} {cz}'.strip())
            for m in mods:
                cand.add(f'{r} {cz} {m}'.strip())
                cand.add(f'{r} {m}'.strip())
    # 2) 네이버 자동완성으로 실제 검색어 확장(있으면 — 더 현실적인 후보)
    for seed in list(cand)[:30]:
        cand |= set(naver_autocomplete(seed))   # 자동완성 API/스크랩
    cand = [c for c in cand if len(c) >= 2][:cap]   # 상한
    # 3) 각 후보 검색 → place_id 가 max_rank 안에 있으면 채택 (RATE 준수)
    found = []
    for kw in cand:
        m = collect_keyword(client, kw, max_rank=max_rank)
        hit = next((x for x in m if x['place_id'] == place_id), None)
        if hit: found.append((kw, hit['rank']))
        time.sleep(3)        # RATE 보호
    found.sort(key=lambda x: x[1])   # 순위순
    return found
```
- 온디맨드 워커(`_analyze_store`)에서 **등록 키워드가 없으면** `discover_keywords()` 로 자동탐색, **있으면** 등록분 우선(+합집합) — 그러면 와벨리 [분석 시작]만 눌러도 됨.
- 후보 생성에 필요한 입력(region/category)은 와벨리 `stores`(category·address) 또는 플레이스 상세에서.
- **와벨리쪽 변경은 단 한 줄**: 자동탐색이 워커에 들어가면 `place-rank.tsx` 의 "키워드 1개+ 필요" 게이트만 빼면 됨(내가 2초 작업). 그전까진 등록 키워드 필요.
- 품질 개선: 채택된 키워드의 자동완성을 한 번 더 돌려 2차 확장 / 경쟁사 상위 키워드 역참조.

### ★★ adlog식 "즉석 자동매칭" = 역색인 구축 (가장 좋은 방식)
adlog가 업체ID 넣으면 바로 키워드가 뜨는 건 실시간 탐색이 아니라 **미리 쌓은 "키워드→업체" 역색인 조회**다. 우리도 같게 하려면:
1. **지역 키워드 universe를 1회 대량 크롤링**: 춘천 관련 키워드 수백 개(업종×지역×수식어+자동완성)를 한 번 수집 → 각 키워드의 상위 N개 업체(place_id) 목록을 DB에 저장. (이미 `keyword_mode.collect_keyword` 가 키워드별 업체 리스트를 줌 — 이걸 keyword_index 테이블에 적재만 하면 됨)
2. **역색인 테이블**: `keyword_index(keyword, place_id, rank, snap_date)`. 한 번 쌓으면 재활용.
3. **업체 분석 = 즉석 역추출**: `select keyword, rank from keyword_index where place_id=? order by rank` → 스크랩 없이 1초. (= adlog 경험)
4. 유지: universe 키워드만 매일/주기적으로 갱신하면 모든 업체가 자동 최신화. (매장 추가돼도 추가 크롤링 거의 불필요)
- 장점: per-매장 후보검색(매장마다 수 분)보다 훨씬 효율 — 지역 색인 1벌로 그 지역 모든 매장 커버.
- RATE: 초기 universe 구축이 큰 작업(수백 키워드)이지만 1회성. 이후엔 universe 갱신만.
- 와벨리 연동: 워커가 `keyword_index` 에서 역추출한 결과를 그대로 `place_rankings` 에 넣으면 됨(스키마 동일). 와벨리 UI 변경 없음.

## 4) 주의
- `analyze_place(place_id)` 는 **키워드를 미리 안 받는다** — adlog처럼 place_id가 노출되는 키워드를 자동 탐색해서 돌려줘야 함(스크린샷처럼 20개 전부). (위 3.5 참고)
- RATE 제한(키워드/요청 간 2.5~5초, 일일 3000) 준수. 워커가 동시에 여러 요청을 처리하지 말 것(한 번에 하나).
- service_role 키 노출 금지. 이 워커는 집 PC에서만 실행.
- 와벨리 폴링은 최대 약 2분 대기 후 "수집기 무응답" 안내 → 그 안에 done 으로 바뀌어야 사용자 화면에 결과가 뜬다.
