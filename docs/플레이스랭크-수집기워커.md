# 플레이스 분석 — 집 PC 수집기 워커 (참고 코드)

> 와벨리 [🔍 분석 시작] 버튼이 실제로 동작하려면, **집 PC 수집기**에 아래 워커가 돌아야 한다.
> 와벨리는 요청만 큐에 넣고, 수집(스크래핑·N지수 계산)은 가정 IP 머신에서만 가능하기 때문.
>
> ⚠️ 이건 **참고 글루 코드**다. 실제 스크래핑/N지수 계산은 너의 기존 수집기 함수를 호출하면 된다
> (`analyze_place(place_id)` 부분). 이 파일은 와벨리 repo에 두는 명세이고, 코드는 260616 수집기 폴더에 둘 것.

## 1) 필요한 것
- 와벨리 Supabase에 `05~08.sql` 적용 완료(특히 `07_place_rank.sql`, `08_place_analysis.sql`).
- 수집기 `.env` 에:
  ```
  SUPABASE_URL=https://tjsjxbbqyrtjblajmose.supabase.co
  SUPABASE_SERVICE_ROLE_KEY=...   # ⚠️ service_role 키. RLS 우회용. 외부 노출 절대 금지(수집기 .env 에만)
  ```
  service_role 키는 Supabase 대시보드 → Project Settings → API → service_role(secret) 에서 복사.

## 2) 워커 루프 (stdlib urllib 기반 — 새 의존성 없음)

```python
# place_collector/scheduler/wavely_worker.py
import os, json, time, urllib.request, urllib.parse

SB_URL = os.environ["SUPABASE_URL"]
SB_KEY = os.environ["SUPABASE_SERVICE_ROLE_KEY"]
HEADERS = {
    "apikey": SB_KEY,
    "Authorization": f"Bearer {SB_KEY}",
    "Content-Type": "application/json",
}

def _req(method, path, params=None, body=None):
    url = f"{SB_URL}/rest/v1/{path}"
    if params:
        url += "?" + urllib.parse.urlencode(params)
    data = json.dumps(body).encode() if body is not None else None
    r = urllib.request.Request(url, data=data, method=method, headers=HEADERS)
    with urllib.request.urlopen(r, timeout=30) as resp:
        txt = resp.read().decode()
        return json.loads(txt) if txt else None

def claim_pending():
    rows = _req("GET", "place_analysis_requests", {
        "status": "eq.pending", "order": "requested_at.asc", "limit": "1",
        "select": "id,store_id",
    })
    if not rows:
        return None
    req = rows[0]
    # running 으로 선점
    _req("PATCH", "place_analysis_requests", {"id": f"eq.{req['id']}"},
         {"status": "running"})
    return req

def store_place_id(store_id):
    rows = _req("GET", "stores", {"id": f"eq.{store_id}", "select": "naver_place_id"})
    return rows[0]["naver_place_id"] if rows else None

def finish(req_id, status, error=None):
    body = {"status": status, "finished_at": "now()"}
    if error:
        body["error"] = str(error)[:500]
    _req("PATCH", "place_analysis_requests", {"id": f"eq.{req_id}"}, body)

def push_results(store_id, place_id, result):
    """result = 너의 기존 수집기 분석 결과(아래 형태로 맞춰서 넘겨줄 것)"""
    today = time.strftime("%Y-%m-%d")
    # 2-1) 키워드별 순위 → place_rankings upsert (store_id,keyword,snap_date 유니크)
    rows = [{
        "store_id": store_id, "keyword": k["keyword"], "rank": k.get("rank"),
        "save_count": result.get("save_count"),
        "visitor_review": result.get("visitor_review"),
        "blog_review": result.get("blog_review"),
        "snap_date": today,
    } for k in result["keywords"]]
    if rows:
        # 충돌 시 갱신: Prefer resolution=merge-duplicates
        h = dict(HEADERS); h["Prefer"] = "resolution=merge-duplicates"
        req = urllib.request.Request(
            f"{SB_URL}/rest/v1/place_rankings?on_conflict=store_id,keyword,snap_date",
            data=json.dumps(rows).encode(), method="POST", headers=h)
        urllib.request.urlopen(req, timeout=30).read()
    # 2-2) 종합 스냅샷 → place_analysis insert
    _req("POST", "place_analysis", None, {
        "store_id": store_id, "place_id": place_id,
        "n1": result.get("n1"), "n2": result.get("n2"), "n3": result.get("n3"),
        "save_count": result.get("save_count"),
        "visitor_review": result.get("visitor_review"),
        "blog_review": result.get("blog_review"),
        "keyword_count": len(result["keywords"]),
        "exposed_count": sum(1 for k in result["keywords"] if k.get("rank")),
        "best_rank": min([k["rank"] for k in result["keywords"] if k.get("rank")], default=None),
        "avg_rank": (lambda rs: round(sum(rs)/len(rs), 2) if rs else None)(
                     [k["rank"] for k in result["keywords"] if k.get("rank")]),
    })

def loop():
    while True:
        try:
            req = claim_pending()
            if not req:
                time.sleep(15); continue
            place_id = store_place_id(req["store_id"])
            if not place_id:
                finish(req["id"], "failed", "연결된 네이버 플레이스 ID가 없어요"); continue
            # ↓↓↓ 여기서 너의 기존 수집기 종합분석 호출 (키워드 자동 발견 포함) ↓↓↓
            result = analyze_place(place_id)   # TODO: 기존 함수로 교체
            #   기대 형태:
            #   { "n1":..., "n2":..., "n3":...,
            #     "save_count":..., "visitor_review":..., "blog_review":...,
            #     "keywords": [ {"keyword":"춘천 닭갈비", "rank": 3}, ... ] }
            push_results(req["store_id"], place_id, result)
            finish(req["id"], "done")
            print(f"[wavely] done store={req['store_id']} kw={len(result['keywords'])}")
        except Exception as e:
            try:
                finish(req["id"], "failed", e)
            except Exception:
                pass
            print("[wavely] error:", e)
        time.sleep(3)  # RATE 보호

if __name__ == "__main__":
    loop()
```

## 3) 매일 자동 push (온디맨드와 별개, 선택)
기존 `scheduler/daily.py` 끝에서, naver_place_id 가 있는 매장들에 대해 `push_results()` 를 그대로 호출하면
매일 새벽 순위가 자동으로 채워진다(버튼 안 눌러도 추이 그래프가 쌓임).

## 4) 주의
- `analyze_place(place_id)` 는 **키워드를 미리 안 받는다** — adlog처럼 place_id가 노출되는 키워드를 자동 탐색해서 돌려줘야 함(스크린샷처럼 20개 전부).
- RATE 제한(키워드/요청 간 2.5~5초, 일일 3000) 준수. 워커가 동시에 여러 요청을 처리하지 말 것(한 번에 하나).
- service_role 키 노출 금지. 이 워커는 집 PC에서만 실행.
- 와벨리 폴링은 최대 약 2분 대기 후 "수집기 무응답" 안내 → 그 안에 done 으로 바뀌어야 사용자 화면에 결과가 뜬다.
